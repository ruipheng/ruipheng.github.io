int main()
{
cout<<"请输入数组总长度"<<endl;

int m=0;

int *arry=new int[count];//根据输入长度新建数组
for(int i=0;i<count;i++)
{arry[i]=i+1;}


int mind=-1;
int k=0,l=0;
for(int k=0;k<count-1;)
{
while(l<m)//每一次向后进一找到m个不为0的数组成序列，把第m个数输出。
{  

if(arry[mind=((mind+1)%count)]>0)
{
l++;
}

}
cout<<"第"<<++k<<"次踢出的人是，第"<<mind<<"个，"<<"数值为："<<arry[(mind%count)]<<endl;    
arry[(mind%count)]=0;

}

for(int i=0;i<count;i++){
if(arry[i]!=0){
cout<<"剩余的人是"<<arry[i]<<endl;
}
}

}
